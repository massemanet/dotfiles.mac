# on the mac

## oneshot
sudo gem update --system
sudo gem update 
sudo gem install vagrant

## per box
NAME=centos-6.2
BOX=http://packages.vstone.eu/vagrant-boxes/centos/6.2/centos-6.2-64bit-puppet-vbox.4.1.10-3.box
vagrant box add $NAME $BOX
mkdir -p ~/boxes/$NAME
cd ~/boxes/$NAME
vagrant init $NAME

## per session
vagrant up
vagrant ssh


# on the centos box

## oneshot dotfiles
cd
sudo mv ~/.ssh /tmp
rm -rf ~/*
rm -rf ~/.*
git clone git://github.com/massemanet/dotfiles.centos.git ~
sudo mv /tmp/.ssh ~

## oneshot normal RPMs
FORGE=rpmforge-release-0.5.2-2.el6.rf.x86_64.rpm 
cd /tmp
wget http://pkgs.repoforge.org/rpmforge-release/$FORGE
sudo rpm -i $FORGE
sudo yum --assumeyes update
sudo yum --assumeyes install git bash-completion man tmux
cd
